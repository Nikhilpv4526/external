step1= mvn archetype:generate -DgroupId=com.example -DartifactId=maven-example -DarchetypeArtifactId=maven-archetype-quickstart -DinteractiveMode=false

step3= app.java
package com.example;

public class App {

    public static void main(String[] args) {
        System.out.println("Hello, Maven");
        System.out.println("This is the simple real-world example....");

        int a = 5;
        int b = 10;

        System.out.println("Sum of " + a + " and " + b + " is " + sum(a, b));
    }

    public static int sum(int x, int y) {
        return x + y;
    }
}


step4= mvn clean install

step5= mvn exec:java -Dexec.mainClass="com.example.App"

step6= gradle init

step7= build.gradle
plugins {
    id 'java'
}

group = 'com.example'
version = '1.0-SNAPSHOT'

repositories {
    mavenCentral()
}

dependencies {
    testImplementation 'junit:junit:4.12'
}

task run(type: JavaExec) {
    main = 'com.example.App'
    classpath = sourceSets.main.runtimeClasspath
}


step8= gradlew build

step9= gradlew run
